16  /* This function is a wrapper and replacement for the photosynthesis code which used

18  is being moved into funtions. */

20  /* psn_struct psn_sun, psn_shade; */
21 
22  /* SUNLIT canopy fraction photosynthesis */
23  /* set the input variables */

84 int photosynthesis(psn_struct* psn)
85 {
86  /*
87  The following variables are assumed to be defined in the psn struct
88  at the time of the function call:
89  c3 (flag) set to 1 for C3 model, 0 for C4 model
90  pa (Pa) atmospheric pressure
91  co2 (ppm) atmospheric [CO2]
92  t (deg C) air temperature
93  lnc (kg Nleaf/m2) leaf N concentration, per unit projected LAI
94  flnr (kg NRub/kg Nleaf) fraction of leaf N in Rubisco
95  ppfd (umol photons/m2/s) PAR flux density, per unit projected LAI
96  g (umol CO2/m2/s/Pa) leaf-scale conductance to CO2, proj area basis
97  dlmr (umol CO2/m2/s) day leaf maint resp, on projected leaf area basis
98 
99  The following variables in psn struct are defined upon function return:
100  Ci (Pa) intercellular [CO2]
101  Ca (Pa) atmospheric [CO2]
102  O2 (Pa) atmospheric [O2]
103  gamma (Pa) CO2 compensation point, in the absence of maint resp.
104  Kc (Pa) MM constant for carboxylation
105  Ko (Pa) MM constant for oxygenation
106  Vmax (umol CO2/m2/s) max rate of carboxylation
107  Jmax (umol electrons/m2/s) max rate electron transport
108  J (umol RuBP/m2/s) rate of RuBP regeneration
109  Av (umol CO2/m2/s) carboxylation limited assimilation
110  Aj (umol CO2/m2/s) RuBP regen limited assimilation
111  A (umol CO2/m2/s) final assimilation rate
112  */
113 
114  /* the weight proportion of Rubisco to its nitrogen content, fnr, is
115  calculated from the relative proportions of the basic amino acids
116  that make up the enzyme, as listed in the Handbook of Biochemistry,
117  Proteins, Vol III, p. 510, which references:
118  Kuehn and McFadden, Biochemistry, 8:2403, 1969 */
119  static double fnr = 7.16; /* kg Rub/kg NRub */
120 
121  /* the following enzyme kinetic constants are from:
122  Woodrow, I.E., and J.A. Berry, 1980. Enzymatic regulation of photosynthetic
123  CO2 fixation in C3 plants. Ann. Rev. Plant Physiol. Plant Mol. Biol.,
124  39:533-594.
125  Note that these values are given in the units used in the paper, and that
126  they are converted to units appropriate to the rest of this function before
127  they are used. */
128  /* I've changed the values for Kc and Ko from the Woodrow and Berry
129  reference, and am now using the values from De Pury and Farquhar,
130  1997. Simple scaling of photosynthesis from leaves to canopies
131  without the errors of big-leaf models. Plant, Cell and Env. 20: 537-557.
132  All other parameters, including the q10's for Kc and Ko are the same
133  as in Woodrow and Berry. */
134  static double Kc25 = 404.0; /* (ubar) MM const carboxylase, 25 deg C */
135  static double q10Kc = 2.1; /* (DIM) Q_10 for Kc */
136  static double Ko25 = 248.0; /* (mbar) MM const oxygenase, 25 deg C */
137  static double q10Ko = 1.2; /* (DIM) Q_10 for Ko */
138  static double act25 = 3.6; /* (umol/mgRubisco/min) Rubisco activity */
139  static double q10act = 2.4; /* (DIM) Q_10 for Rubisco activity */
140  static double pabs = 0.85; /* (DIM) fPAR effectively absorbed by PSII */
141 
142  /* local variables */
143  int ok=1;
144  double t; /* (deg C) temperature */
145  double tk; /* (K) absolute temperature */
146  double Kc; /* (Pa) MM constant for carboxylase reaction */
147  double Ko; /* (Pa) MM constant for oxygenase reaction */
148  double act; /* (umol CO2/kgRubisco/s) Rubisco activity */
149  double Jmax; /* (umol/m2/s) max rate electron transport */
150  double ppe; /* (mol/mol) photons absorbed by PSII per e- transported */
151  double Vmax, J, gamma, Ca, Rd, O2, g;
152  double a,b,c,det;
153  double Av,Aj, A;
154 
155 
156  /* begin by assigning local variables */
157  g = psn->g;
158  t = psn->t;
159  tk = t + 273.15;
160  Rd = psn->dlmr;
161 
162  /* convert atmospheric CO2 from ppm --> Pa */
163  Ca = psn->co2 * psn->pa / 1e6;
164 
165  /* set parameters for C3 vs C4 model */
166  if (psn->c3)
167  {
168  ppe = 2.6;
169  }
170  else /* C4 */
171  {
172  ppe = 3.5;
173  Ca *= 10.0;
174  }
175  psn->Ca = Ca;
176 
177  /* calculate atmospheric O2 in Pa, assumes 21% O2 by volume */
178  psn->O2 = O2 = 0.21 * psn->pa;
179 
180  /* correct kinetic constants for temperature, and do unit conversions */
181  Ko = Ko25 * pow(q10Ko, (t-25.0)/10.0);
182  psn->Ko = Ko = Ko * 100.0; /* mbar --> Pa */
183  if (t > 15.0)
184  {
185  Kc = Kc25 * pow(q10Kc, (t-25.0)/10.0);
186  act = act25 * pow(q10act, (t-25.0)/10.0);
187  }
188  else
189  {
190  Kc = Kc25 * pow(1.8*q10Kc, (t-15.0)/10.0) / q10Kc;
191  act = act25 * pow(1.8*q10act, (t-15.0)/10.0) / q10act;
192  }
193  psn->Kc = Kc = Kc * 0.10; /* ubar --> Pa */
194  act = act * 1e6 / 60.0; /* umol/mg/min --> umol/kg/s */
195 
196  /* calculate gamma (Pa), assumes Vomax/Vcmax = 0.21 */
197  psn->gamma = gamma = 0.5 * 0.21 * Kc * psn->O2 / Ko;
198 
199  /* calculate Vmax from leaf nitrogen data and Rubisco activity */
200 
201  /* kg Nleaf kg NRub kg Rub umol umol
202  -------- X ------- X ------- X --------- = --------
203  m2 kg Nleaf kg NRub kg RUB * s m2 * s
204 
205  (lnc) X (flnr) X (fnr) X (act) = (Vmax)
206  */
207  psn->Vmax = Vmax = psn->lnc * psn->flnr * fnr * act;
208 
209  /* calculate Jmax = f(Vmax), reference:
210  Wullschleger, S.D., 1993. Biochemical limitations to carbon assimilation
211  in C3 plants - A retrospective analysis of the A/Ci curves from
212  109 species. Journal of Experimental Botany, 44:907-920.
213  */
214  psn->Jmax = Jmax = 2.1*Vmax;
215 
216  /* calculate J = f(Jmax, ppfd), reference:
217  de Pury and Farquhar 1997
218  Plant Cell and Env.
219  */
220  a = 0.7;
221  b = -Jmax - (psn->ppfd*pabs/ppe);
222  c = Jmax * psn->ppfd*pabs/ppe;
223  psn->J = J = (-b - sqrt(b*b - 4.0*a*c))/(2.0*a);
224 
225  /* solve for Av and Aj using the quadratic equation, substitution for Ci
226  from A = g(Ca-Ci) into the equations from Farquhar and von Caemmerer:
227 
228  Vmax (Ci - gamma)
229  Av = ------------------- - Rd
230  Ci + Kc (1 + O2/Ko)
231 
232 
233  J (Ci - gamma)
234  Aj = ------------------- - Rd
235  4.5 Ci + 10.5 gamma
236  */
237 
238  /* quadratic solution for Av */
239  a = -1.0/g;
240  b = Ca + (Vmax - Rd)/g + Kc*(1.0 + O2/Ko);
241  c = Vmax*(gamma - Ca) + Rd*(Ca + Kc*(1.0 + O2/Ko));
242 
243  if ((det = b*b - 4.0*a*c) < 0.0)
244  {
245  bgc_printf(BV_ERROR, "negative root error in psn routine\n");
246  ok=0;
247  }
248 
249  psn->Av = Av = (-b + sqrt(det)) / (2.0*a);
250 
251  /* quadratic solution for Aj */
252  a = -4.5/g;
253  b = 4.5*Ca + 10.5*gamma + J/g - 4.5*Rd/g;
254  c = J*(gamma - Ca) + Rd*(4.5*Ca + 10.5*gamma);
255 
256  if ((det = b*b - 4.0*a*c) < 0.0)
257  {
258  bgc_printf(BV_ERROR, "negative root error in psn routine\n");
259  ok=0;
260  }
261 
262  psn->Aj = Aj = (-b + sqrt(det)) / (2.0*a);
263 
264  /* estimate A as the minimum of (Av,Aj) */
265  if (Av < Aj) A = Av;
266  else A = Aj;
267  psn->A = A;
268  bgc_printf(BV_DIAG, "psn->A: %f, A: %f\n", psn->A, A);
269  psn->Ci = Ca - (A/g);
270 
271  return (!ok);
272 }

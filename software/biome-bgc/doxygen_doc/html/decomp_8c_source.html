16 int decomp(double tsoil, const epconst_struct* epc, epvar_struct* epv,
17 const siteconst_struct* sitec, cstate_struct* cs, cflux_struct* cf,
18 nstate_struct* ns, nflux_struct* nf, ntemp_struct* nt)
19 {
20  int ok=1;
21  double rate_scalar, t_scalar, w_scalar;
22  double tk;
23  double minpsi, maxpsi;
24  double rfl1s1, rfl2s2,rfl4s3,rfs1s2,rfs2s3,rfs3s4;
25  double kl1_base,kl2_base,kl4_base,ks1_base,ks2_base,ks3_base,ks4_base,kfrag_base;
26  double kl1,kl2,kl4,ks1,ks2,ks3,ks4,kfrag;
27  double cn_l1,cn_l2,cn_l4,cn_s1,cn_s2,cn_s3,cn_s4;
28  double cwdc_loss;
29  double plitr1c_loss, plitr2c_loss, plitr4c_loss;
30  double psoil1c_loss, psoil2c_loss, psoil3c_loss, psoil4c_loss;
31  double pmnf_l1s1,pmnf_l2s2,pmnf_l4s3,pmnf_s1s2,pmnf_s2s3,pmnf_s3s4,pmnf_s4;
32  double potential_immob,mineralized;
33  int nlimit;
34  double ratio;
35 
36  /* calculate the rate constant scalar for soil temperature,
37  assuming that the base rate constants are assigned for non-moisture
38  limiting conditions at 25 C. The function used here is taken from
39  Lloyd, J., and J.A. Taylor, 1994. On the temperature dependence of
40  soil respiration. Functional Ecology, 8:315-323.
41  This equation is a modification of their eqn. 11, changing the base
42  temperature from 10 C to 25 C, since most of the microcosm studies
43  used to get the base decomp rates were controlled at 25 C. */
44  if (tsoil < -10.0)
45  {
46  /* no decomp processes for tsoil < -10.0 C */
47  t_scalar = 0.0;
48  }
49  else
50  {
51  tk = tsoil + 273.15;
52  t_scalar = exp(308.56*((1.0/71.02)-(1.0/(tk-227.13))));
53  }
54 
55  /* calculate the rate constant scalar for soil water content.
56  Uses the log relationship with water potential given in
57  Andren, O., and K. Paustian, 1987. Barley straw decomposition in the field:
58  a comparison of models. Ecology, 68(5):1190-1200.
59  and supported by data in
60  Orchard, V.A., and F.J. Cook, 1983. Relationship between soil respiration
61  and soil moisture. Soil Biol. Biochem., 15(4):447-453.
62  */
63  /* set the maximum and minimum values for water potential limits (MPa) */
64  minpsi = -10.0;
65  maxpsi = sitec->psi_sat;
66  if (epv->psi < minpsi)
67  {
68  /* no decomp below the minimum soil water potential */
69  w_scalar = 0.0;
70  }
71  else if (epv->psi > maxpsi)
72  {
73  /* this shouldn't ever happen, but just in case... */
74  w_scalar = 1.0;
75  }
76  else
77  {
78  w_scalar = log(minpsi/epv->psi)/log(minpsi/maxpsi);
79  }
80 
81  /* calculate the final rate scalar as the product of the temperature and
82  water scalars */
83  rate_scalar = w_scalar * t_scalar;
84 
85  /* assign output variables */
86  epv->t_scalar = t_scalar;
87  epv->w_scalar = w_scalar;
88  epv->rate_scalar = rate_scalar;
89 
90  /* calculate compartment C:N ratios */
91  if (ns->litr1n > 0.0) cn_l1 = cs->litr1c/ns->litr1n;
92  if (ns->litr2n > 0.0) cn_l2 = cs->litr2c/ns->litr2n;
93  if (ns->litr4n > 0.0) cn_l4 = cs->litr4c/ns->litr4n;
94  cn_s1 = SOIL1_CN;
95  cn_s2 = SOIL2_CN;
96  cn_s3 = SOIL3_CN;
97  cn_s4 = SOIL4_CN;
98 
99  /* respiration fractions for fluxes between compartments */
100  rfl1s1 = RFL1S1;
101  rfl2s2 = RFL2S2;
102  rfl4s3 = RFL4S3;
103  rfs1s2 = RFS1S2;
104  rfs2s3 = RFS2S3;
105  rfs3s4 = RFS3S4;
106 
107  /* calculate the corrected rate constants from the rate scalar and their
108  base values. All rate constants are (1/day) */
109  kl1_base = KL1_BASE; /* labile litter pool */
110  kl2_base = KL2_BASE; /* cellulose litter pool */
111  kl4_base = KL4_BASE; /* lignin litter pool */
112  ks1_base = KS1_BASE; /* fast microbial recycling pool */
113  ks2_base = KS2_BASE; /* medium microbial recycling pool */
114  ks3_base = KS3_BASE; /* slow microbial recycling pool */
115  ks4_base = KS4_BASE; /* recalcitrant SOM (humus) pool */
116  kfrag_base = KFRAG_BASE; /* physical fragmentation of coarse woody debris */
117  kl1 = kl1_base * rate_scalar;
118  kl2 = kl2_base * rate_scalar;
119  kl4 = kl4_base * rate_scalar;
120  ks1 = ks1_base * rate_scalar;
121  ks2 = ks2_base * rate_scalar;
122  ks3 = ks3_base * rate_scalar;
123  ks4 = ks4_base * rate_scalar;
124  kfrag = kfrag_base * rate_scalar;
125 
126  /* woody vegetation type fluxes */
127  if (epc->woody)
128  {
129  /* calculate the flux from CWD to litter lignin and cellulose
130  compartments, due to physical fragmentation */
131  cwdc_loss = kfrag * cs->cwdc;
132  cf->cwdc_to_litr2c = cwdc_loss * epc->deadwood_fucel;
133  cf->cwdc_to_litr3c = cwdc_loss * epc->deadwood_fscel;
134  cf->cwdc_to_litr4c = cwdc_loss * epc->deadwood_flig;
135  nf->cwdn_to_litr2n = cf->cwdc_to_litr2c/epc->deadwood_cn;
136  nf->cwdn_to_litr3n = cf->cwdc_to_litr3c/epc->deadwood_cn;
137  nf->cwdn_to_litr4n = cf->cwdc_to_litr4c/epc->deadwood_cn;
138  }
139 
140  /* initialize the potential loss and mineral N flux variables */
141  plitr1c_loss = plitr2c_loss = plitr4c_loss = 0.0;
142  psoil1c_loss = psoil2c_loss = psoil3c_loss = psoil4c_loss = 0.0;
143  pmnf_l1s1 = pmnf_l2s2 = pmnf_l4s3 = 0.0;
144  pmnf_s1s2 = pmnf_s2s3 = pmnf_s3s4 = pmnf_s4 = 0.0;
145 
146  /* calculate the non-nitrogen limited fluxes between litter and
147  soil compartments. These will be ammended for N limitation if it turns
148  out the potential gross immobilization is greater than potential gross
149  mineralization. */
150  /* 1. labile litter to fast microbial recycling pool */
151  if (cs->litr1c > 0.0)
152  {
153  plitr1c_loss = kl1 * cs->litr1c;
154  if (ns->litr1n > 0.0) ratio = cn_s1/cn_l1;
155  else ratio = 0.0;
156  pmnf_l1s1 = (plitr1c_loss * (1.0 - rfl1s1 - (ratio)))/cn_s1;
157  }
158 
159  /* 2. cellulose litter to medium microbial recycling pool */
160  if (cs->litr2c > 0.0)
161  {
162  plitr2c_loss = kl2 * cs->litr2c;
163  if (ns->litr2n > 0.0) ratio = cn_s2/cn_l2;
164  else ratio = 0.0;
165  pmnf_l2s2 = (plitr2c_loss * (1.0 - rfl2s2 - (ratio)))/cn_s2;
166  }
167 
168  /* 3. lignin litter to slow microbial recycling pool */
169  if (cs->litr4c > 0.0)
170  {
171  plitr4c_loss = kl4 * cs->litr4c;
172  if (ns->litr4n > 0.0) ratio = cn_s3/cn_l4;
173  else ratio = 0.0;
174  pmnf_l4s3 = (plitr4c_loss * (1.0 - rfl4s3 - (ratio)))/cn_s3;
175  }
176 
177  /* 4. fast microbial recycling pool to medium microbial recycling pool */
178  if (cs->soil1c > 0.0)
179  {
180  psoil1c_loss = ks1 * cs->soil1c;
181  pmnf_s1s2 = (psoil1c_loss * (1.0 - rfs1s2 - (cn_s2/cn_s1)))/cn_s2;
182  }
183 
184  /* 5. medium microbial recycling pool to slow microbial recycling pool */
185  if (cs->soil2c > 0.0)
186  {
187  psoil2c_loss = ks2 * cs->soil2c;
188  pmnf_s2s3 = (psoil2c_loss * (1.0 - rfs2s3 - (cn_s3/cn_s2)))/cn_s3;
189  }
190 
191  /* 6. slow microbial recycling pool to recalcitrant SOM pool */
192  if (cs->soil3c > 0.0)
193  {
194  psoil3c_loss = ks3 * cs->soil3c;
195  pmnf_s3s4 = (psoil3c_loss * (1.0 - rfs3s4 - (cn_s4/cn_s3)))/cn_s4;
196  }
197 
198  /* 7. mineralization of recalcitrant SOM */
199  if (cs->soil4c > 0.0)
200  {
201  psoil4c_loss = ks4 * cs->soil4c;
202  pmnf_s4 = -psoil4c_loss/cn_s4;
203  }
204 
205  /* determine if there is sufficient mineral N to support potential
206  immobilization. Immobilization fluxes are positive, mineralization fluxes
207  are negative */
208  nlimit = 0;
209  potential_immob = 0.0;
210  mineralized = 0.0;
211  if (pmnf_l1s1 > 0.0) potential_immob += pmnf_l1s1;
212  else mineralized += -pmnf_l1s1;
213  if (pmnf_l2s2 > 0.0) potential_immob += pmnf_l2s2;
214  else mineralized += -pmnf_l2s2;
215  if (pmnf_l4s3 > 0.0) potential_immob += pmnf_l4s3;
216  else mineralized += -pmnf_l4s3;
217  if (pmnf_s1s2 > 0.0) potential_immob += pmnf_s1s2;
218  else mineralized += -pmnf_s1s2;
219  if (pmnf_s2s3 > 0.0) potential_immob += pmnf_s2s3;
220  else mineralized += -pmnf_s2s3;
221  if (pmnf_s3s4 > 0.0) potential_immob += pmnf_s3s4;
222  else mineralized += -pmnf_s3s4;
223  mineralized += -pmnf_s4;
224 
225  /* save the potential fluxes until plant demand has been assessed,
226  to allow competition between immobilization fluxes and plant growth
227  demands */
228  nt->mineralized = mineralized;
229  nt->potential_immob = potential_immob;
230  nt->plitr1c_loss = plitr1c_loss;
231  nt->pmnf_l1s1 = pmnf_l1s1;
232  nt->plitr2c_loss = plitr2c_loss;
233  nt->pmnf_l2s2 = pmnf_l2s2;
234  nt->plitr4c_loss = plitr4c_loss;
235  nt->pmnf_l4s3 = pmnf_l4s3;
236  nt->psoil1c_loss = psoil1c_loss;
237  nt->pmnf_s1s2 = pmnf_s1s2;
238  nt->psoil2c_loss = psoil2c_loss;
239  nt->pmnf_s2s3 = pmnf_s2s3;
240  nt->psoil3c_loss = psoil3c_loss;
241  nt->pmnf_s3s4 = pmnf_s3s4;
242  nt->psoil4c_loss = psoil4c_loss;
243  nt->kl4 = kl4;
244 
245  /* store the day's gross mineralization */
246  epv->daily_gross_nmin = mineralized ;
247 
248  return (!ok);
249 }

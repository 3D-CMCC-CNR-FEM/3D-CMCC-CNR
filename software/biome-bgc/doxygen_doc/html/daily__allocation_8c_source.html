84  /* no allocation when the daily C balance is negative */

87  /* test for cpool deficit */
88  if (cs->cpool < 0.0)
89  {
90  /* running a deficit in cpool, so the first priority
91  is to let some of today's available C accumulate in cpool. The actual
92  accumulation in the cpool is resolved in day_carbon_state(). */
93  /* first determine how much of the deficit should be recovered today */
94  cpool_recovery = -cs->cpool/DAYSCRECOVER;
95  if (cpool_recovery < avail_c)
96  {
97  /* potential recovery of cpool deficit is less than the available
98  carbon for the day, so aleviate cpool deficit
99  and use the rest of the available carbon for
100  new growth and storage. Remember that fluxes in
101  and out of the cpool are reconciled at the end
102  of the daily loop, so for now, just keep track
103  of the amount of daily GPP-MR that is not needed
104  to restore a negative cpool. */
105  avail_c -= cpool_recovery;
106  }
107  else
108  {
109  /* cpool deficit is >= available C, so all of the
110  daily GPP, if any, is used to alleviate negative cpool */
111  avail_c = 0.0;
112  }
113  } /* end if negative cpool */

116  f1 = epc->alloc_frootc_leafc;
117  f2 = epc->alloc_crootc_stemc;

119  f4 = epc->alloc_newlivewoodc_newwoodc;

123  cnfr = epc->froot_cn;
124  cnlw = epc->livewood_cn;
125  cndw = epc->deadwood_cn;

130  demand */

132  {

135  (f3*(1.0-f4)*(1.0+f2))/cndw);
136  }
137  else

140  n_allometry = (1.0/cnl + f1/cnfr);
141  }

147  sum_ndemand = plant_ndemand + nt->potential_immob;
148

151  if (mode == MODE_SPINUP)

154  {

157  ns->ndep_src += dif * naddfrac;
158  }

166  nlimit = 0;

169  N to meet the plant demand */

203  proportion lost in the decomposition pathways. */

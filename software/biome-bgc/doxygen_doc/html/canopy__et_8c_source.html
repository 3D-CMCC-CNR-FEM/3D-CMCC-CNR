51  /* calculate leaf- and canopy-level conductances to water vapor and
52  sensible heat fluxes */
53 
54  /* leaf boundary-layer conductance */
55  gl_bl = epc->gl_bl * gcorr;
56 
57  /* leaf cuticular conductance */
58  gl_c = epc->gl_c * gcorr;
59 
60  /* leaf stomatal conductance: first generate multipliers, then apply them
61  to maximum stomatal conductance */
62  /* calculate stomatal conductance radiation multiplier:
63  *** NOTE CHANGE FROM BIOME-BGC CODE ***
64  The original Biome-BGC formulation follows the arguments in
65  Rastetter, E.B., A.W. King, B.J. Cosby, G.M. Hornberger,
66  R.V. O'Neill, and J.E. Hobbie, 1992. Aggregating fine-scale
67  ecological knowledge to model coarser-scale attributes of
68  ecosystems. Ecological Applications, 2:55-70.
69 
70  gmult->max = (gsmax/(k*lai))*log((gsmax+rad)/(gsmax+(rad*exp(-k*lai))))
71 
72  I'm using a much simplified form, which doesn't change relative shape
73  as gsmax changes. See Korner, 1995.
74  */
75  /* photosynthetic photon flux density conductance control */
76  m_ppfd_sun = metv->ppfd_per_plaisun/(PPFD50 + metv->ppfd_per_plaisun);
77  m_ppfd_shade = metv->ppfd_per_plaishade/(PPFD50 + metv->ppfd_per_plaishade);
78 
79  /* soil-leaf water potential multiplier */
80  if (psi > psi_open) /* no water stress */
81  m_psi = 1.0;
82  else
83  if (psi <= psi_close) /* full water stress */
84  {
85  m_psi = 0.0;
86  }
87  else /* partial water stress */
88  m_psi = (psi_close - psi) / (psi_close - psi_open);
89 
90  /* CO2 multiplier */
91  m_co2 = 1.0;
92 
93  /* freezing night minimum temperature multiplier */
94  if (tmin > 0.0) /* no effect */
95  m_tmin = 1.0;
96  else
97  if (tmin < -8.0) /* full tmin effect */
98  m_tmin = 0.0;
99  else /* partial reduction (0.0 to -8.0 C) */
100  m_tmin = 1.0 + (0.125 * tmin);
101 
102  /* vapor pressure deficit multiplier, vpd in Pa */
103  if (vpd < vpd_open) /* no vpd effect */
104  m_vpd = 1.0;
105  else
106  if (vpd > vpd_close) /* full vpd effect */
107  m_vpd = 0.0;
108  else /* partial vpd effect */
109  m_vpd = (vpd_close - vpd) / (vpd_close - vpd_open);
110 
111  /* apply all multipliers to the maximum stomatal conductance */
112  m_final_sun = m_ppfd_sun * m_psi * m_co2 * m_tmin * m_vpd;
113  if (m_final_sun < 0.00000001) m_final_sun = 0.00000001;
114  m_final_shade = m_ppfd_shade * m_psi * m_co2 * m_tmin * m_vpd;
115  if (m_final_shade < 0.00000001) m_final_shade = 0.00000001;
116  gl_s_sun = epc->gl_smax * m_final_sun * gcorr;
117  gl_s_shade = epc->gl_smax * m_final_shade * gcorr;
118 
119  /* calculate leaf-and canopy-level conductances to water vapor and
120  sensible heat fluxes, to be used in Penman-Monteith calculations of
121  canopy evaporation and canopy transpiration. */
122 
123  /* Leaf conductance to evaporated water vapor, per unit projected LAI */
124  gl_e_wv = gl_bl;
125 
126  /* Leaf conductance to transpired water vapor, per unit projected
127  LAI. This formula is derived from stomatal and cuticular conductances
128  in parallel with each other, and both in series with leaf boundary
129  layer conductance. */
130  gl_t_wv_sun = (gl_bl * (gl_s_sun + gl_c)) / (gl_bl + gl_s_sun + gl_c);
131  gl_t_wv_shade = (gl_bl * (gl_s_shade + gl_c)) / (gl_bl + gl_s_shade + gl_c);
132 
133  /* Leaf conductance to sensible heat, per unit all-sided LAI */
134  gl_sh = gl_bl;
135 
136  /* Canopy conductance to evaporated water vapor */
137  gc_e_wv = gl_e_wv * proj_lai;
138 
139  /* Canopy conductane to sensible heat */
140  gc_sh = gl_sh * proj_lai;
141 
142  cwe = trans = 0.0;
143 
144  /* Assign values in pmet_in that don't change */
145  pmet_in.ta = tday;
146  pmet_in.pa = metv->pa;
147  pmet_in.vpd = vpd;

149  /* Canopy evaporation, if any water was intercepted */
150  /* Calculate Penman-Monteith evaporation, given the canopy conductances to
151  evaporated water and sensible heat. Calculate the time required to
152  evaporate all the canopy water at the daily average conditions, and
153  subtract that time from the daylength to get the effective daylength for
154  transpiration. */

156  {
157  /* assign appropriate resistance and radiation for pmet_in */

273 int penmon(const pmet_struct* in, int out_flag, double* et)
274 {
275  /*
276  Combination equation for determining evaporation and transpiration.
277 
278  For output in units of (kg/m2/s) out_flag = 0
279  For output in units of (W/m2) out_flag = 1
280 
281  INPUTS:
282  in->ta (deg C) air temperature
283  in->pa (Pa) air pressure
284  in->vpd (Pa) vapor pressure deficit
285  in->irad (W/m2) incident radient flux density
286  in->rv (s/m) resistance to water vapor flux
287  in->rh (s/m) resistance to sensible heat flux
288 
289  INTERNAL VARIABLES:
290  rho (kg/m3) density of air
291  CP (J/kg/degC) specific heat of air
292  lhvap (J/kg) latent heat of vaporization of water
293  s (Pa/degC) slope of saturation vapor pressure vs T curve
294 
295  OUTPUT:
296  et (kg/m2/s) water vapor mass flux density (flag=0)
297  et (W/m2) latent heat flux density (flag=1)
298  */
299 
300  int ok=1;
301  double ta;
302  double rho,lhvap,s;
303  double t1,t2,pvs1,pvs2,e,tk;
304  double rr,rh,rhr,rv;
305  double dt = 0.2; /* set the temperature offset for slope calculation */
306 
307  /* assign ta (Celsius) and tk (Kelvins) */
308  ta = in->ta;
309  tk = ta + 273.15;
310 
311  /* calculate density of air (rho) as a function of air temperature */
312  rho = 1.292 - (0.00428 * ta);
313 
314  /* calculate resistance to radiative heat transfer through air, rr */
315  rr = rho * CP / (4.0 * SBC * (tk*tk*tk));
316 
317  /* resistance to convective heat transfer */
318  rh = in->rh;
319 
320  /* resistance to latent heat transfer */
321  rv = in->rv;
322 
323  /* calculate combined resistance to convective and radiative heat transfer,
324  parallel resistances : rhr = (rh * rr) / (rh + rr) */
325  rhr = (rh * rr) / (rh + rr);
326 
327  /* calculate latent heat of vaporization as a function of ta */
328  lhvap = 2.5023e6 - 2430.54 * ta;
329 
330  /* calculate temperature offsets for slope estimate */
331  t1 = ta+dt;
332  t2 = ta-dt;
333 
334  /* calculate saturation vapor pressures at t1 and t2 */
335  pvs1 = 610.7 * exp(17.38 * t1 / (239.0 + t1));
336  pvs2 = 610.7 * exp(17.38 * t2 / (239.0 + t2));
337 
338  /* calculate slope of pvs vs. T curve, at ta */
339  s = (pvs1-pvs2) / (t1-t2);
340 
341  /* calculate evaporation, in W/m^2 */
342  e = ( ( s * in->irad ) + ( rho * CP * in->vpd / rhr ) ) /
343  ( ( ( in->pa * CP * rv ) / ( lhvap * EPS * rhr ) ) + s );
344 
345  /* return either W/m^2 or kg/m^2/s, depending on out_flag */
346  if (out_flag)
347  *et = e;
348 
349  if (!out_flag)
350  *et = e / lhvap;
351 
352  return (!ok);
353 }

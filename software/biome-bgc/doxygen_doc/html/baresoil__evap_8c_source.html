13 int baresoil_evap(const metvar_struct* metv, wflux_struct* wf, double* dsr_ptr)
14 {
15  int ok=1;
16  double dsr; /* number of days since rain */
17  double rbl; /* (m/s) boundary layer resistance */
18  double pot_evap; /* (kg/m2/s) potential evaporation (daytime) */
19  double evap; /* (kg/m2/s) actual evaporation (daytime) */
20  pmet_struct pmet_in; /* input structure for penmon function */
21  double rcorr; /* correction factor for temp and pressure */
22  double ratio; /* actual/potential evaporation for dry day */
23 
24  /* assign days since rain */
25  dsr = *dsr_ptr;
26 
27  /* correct conductances for temperature and pressure based on Jones (1992)
28  with standard conditions assumed to be 20 deg C, 101300 Pa */
29  rcorr = 1.0/(pow((metv->tday+273.15)/293.15, 1.75) * 101300/metv->pa);
30 
31  /* new bare-soil evaporation routine */
32  /* first calculate potential evaporation, assuming the resistance
33  for vapor transport is equal to the resistance for sensible heat
34  transport. That is, no additional resistance for vapor transport to
35  the soil surface. This represents evaporation from a wet surface with
36  a specified aerodynamic resistance (= boundary layer resistance).
37  The aerodynamic resistance is for now set as a constant, and is
38  taken from observations over bare soil in tiger-bush in south-west
39  Niger: rbl = 107 s m-1 (Wallace and Holwill, 1997). */
40  rbl = 107.0 * rcorr;
41 
42  /* fill the pmet_in structure */
43  pmet_in.ta = metv->tday;
44  pmet_in.pa = metv->pa;
45  pmet_in.vpd = metv->vpd;
46  pmet_in.irad = metv->swtrans;
47  pmet_in.rv = rbl;
48  pmet_in.rh = rbl;
49 
50  /* calculate pot_evap in kg/m2/s */
51  penmon(&pmet_in, 0, &pot_evap);
52 
53  /* convert to daily total kg/m2 */
54  pot_evap *= metv->dayl;
55 
56  /* consider only the precipitation flux reaching the soil */
57  /* check for precipitation >= potential evaporation */
58  if (wf->prcp_to_soilw >= pot_evap)
59  {
60  /* reset days-since-rain parameter */
61  dsr = 0.0;
62 
63  /* soil evaporation proceeds at potential rate */
64  evap = 0.6 * pot_evap;
65  }
66  else
67  {
68  /* increment the days since rain */
69  dsr += 1.0;
70 
71  /* calculate the realized proportion of potential evaporation
72  as a function of the days since rain */
73  ratio = 0.3/pow(dsr,2.0);
74 
75  /* calculate evaporation for dry days */
76  evap = ratio * pot_evap;
77 
78  /* for rain events that are smaller than required to reset dsr
79  counter, but larger than dry-day evaporation, all rain is evaporated.
80  In this case, do not advance the drying curve counter.
81  For rain events that are too small to trigger dsr reset, and which
82  are smaller than dry-day evap, there will be more evaporation than
83  rainfall. In this case the drying curve counter is advanced. */
84  if (wf->prcp_to_soilw > evap)
85  {
86  evap = wf->prcp_to_soilw;
87  dsr -= 1.0;
88  }
89  }
90 
91  wf->soilw_evap = evap;
92  *dsr_ptr = dsr;
93 
94  return(!ok);
95 }

13 int maint_resp(const cstate_struct* cs, const nstate_struct* ns,
14 const epconst_struct* epc, const metvar_struct* metv, cflux_struct* cf,
15 epvar_struct* epv)
16 {
17  /*
18  maintenance respiration routine
19  Uses reference values at 20 deg C and an empirical relationship between
20  tissue N content and respiration rate given in:
21 
22  Ryan, M.G., 1991. Effects of climate change on plant respiration.
23  Ecological Applications, 1(2):157-167.
24 
25  Uses the same value of Q_10 (2.0) for all compartments, leaf, stem,
26  coarse and fine roots.
27 
28  From Ryan's figures and regressions equations, the maintenance respiration
29  in kgC/day per kg of tissue N is:
30  mrpern = 0.218 (kgC/kgN/d)
31 
32  Leaf maintenance respiration is calculated separately for day and
33  night, since the PSN routine needs the daylight value.
34 
35  Leaf and fine root respiration are dependent on phenology.
36  */
37 
38  int ok=1;
39  double t1;
40  double q10 = 2.0;
41  double mrpern = 0.218;
42  double exponent;
43  double n_area_sun, n_area_shade, dlmr_area_sun, dlmr_area_shade;
44 
45  /* leaf day and night maintenance respiration when leaves on */
46  if (cs->leafc)
47  {
48  t1 = ns->leafn * mrpern;
49 
50  /* leaf, day */
51  exponent = (metv->tday - 20.0) / 10.0;
52  cf->leaf_day_mr = t1 * pow(q10, exponent) * metv->dayl / 86400.0;
53 
54  /* for day respiration, also determine rates of maintenance respiration
55  per unit of projected leaf area in the sunlit and shaded portions of
56  the canopy, for use in the photosynthesis routine */
57  /* first, calculate the mass of N per unit of projected leaf area
58  in each canopy fraction (kg N/m2 projected area) */
59  n_area_sun = 1.0/(epv->sun_proj_sla * epc->leaf_cn);
60  n_area_shade = 1.0/(epv->shade_proj_sla * epc->leaf_cn);
61  /* convert to respiration flux in kg C/m2 projected area/day, and
62  correct for temperature */
63  dlmr_area_sun = n_area_sun * mrpern * pow(q10, exponent);
64  dlmr_area_shade = n_area_shade * mrpern * pow(q10, exponent);
65  /* finally, convert from mass to molar units, and from a daily rate to
66  a rate per second */
67  epv->dlmr_area_sun = dlmr_area_sun/(86400.0 * 12.011e-9);
68  epv->dlmr_area_shade = dlmr_area_shade/(86400.0 * 12.011e-9);
69 
70  /* leaf, night */
71  exponent = (metv->tnight - 20.0) / 10.0;
72  cf->leaf_night_mr = t1 * pow(q10, exponent) *
73  (86400.0 - metv->dayl) / 86400.0;
74  }
75  else /* no leaves on */
76  {
77  cf->leaf_day_mr = 0.0;
78  epv->dlmr_area_sun = 0.0;
79  epv->dlmr_area_shade = 0.0;
80  cf->leaf_night_mr = 0.0;
81  }
82 
83  /* fine root maintenance respiration when fine roots on */
84  /* ammended to consider only the specified n concentration,
85  to avoid excessive MR with n-loading to fine roots */
86  if (cs->frootc)
87  {
88  exponent = (metv->tsoil - 20.0) / 10.0;
89  t1 = pow(q10, exponent);
90  cf->froot_mr = ns->frootn * mrpern * t1;
91  }
92  else /* no fine roots on */
93  cf->froot_mr = 0.0;
94 
95  /* TREE-specific fluxes */
96  if (epc->woody)
97  {
98  /* live stem maintenance respiration */
99  exponent = (metv->tavg - 20.0) / 10.0;
100  t1 = pow(q10, exponent);
101  cf->livestem_mr = ns->livestemn * mrpern * t1;
102 
103  /* live coarse root maintenance respiration */
104  exponent = (metv->tsoil - 20.0) / 10.0;
105  t1 = pow(q10, exponent);
106  cf->livecroot_mr = ns->livecrootn * mrpern * t1;
107  }
108 
109  return (!ok);
110 }

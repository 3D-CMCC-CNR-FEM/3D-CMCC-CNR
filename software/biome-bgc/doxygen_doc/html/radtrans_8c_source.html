16 metvar_struct* metv, epvar_struct* epv, double albedo)
17 {
18  /* calculate the projected leaf area and SLA for sun and shade fractions
19  and the canopy transmission and absorption of shortwave radiation
20  based on the Beer's Law assumption of radiation attenuation as a
21  function of projected LAI.
22  */
23 
24  int ok=1;
25  double proj_lai;
26  double albedo_sw, albedo_par;
27  double sw,par;
28  double swabs, swtrans;
29  double parabs;
30  double k;
31  double k_sw, k_par;
32  double swabs_plaisun, swabs_plaishade;
33  double swabs_per_plaisun, swabs_per_plaishade;
34  double parabs_plaisun, parabs_plaishade;
35  double parabs_per_plaisun, parabs_per_plaishade;
36 
37  /* The following equations estimate the albedo and extinction
38  coefficients for the shortwave and PAR spectra from the values given for the
39  entire shortwave range (from Jones, H.G., 1992. Plants and Microclimate,
40  2nd Edition. Cambridge University Press. pp. 30-38.) These conversions
41  are approximated from the information given in Jones.
42  */
43 
44  if (cs->leafc > 0.0)
45  {
46  /* Calculate whole-canopy projected and all-sided LAI */
47  epv->proj_lai = cs->leafc * epc->avg_proj_sla;
48  epv->all_lai = epv->proj_lai * epc->lai_ratio;
49 
50  /* Calculate projected LAI for sunlit and shaded canopy portions */
51  epv->plaisun = 1.0 - exp(-epv->proj_lai);
52  epv->plaishade = epv->proj_lai - epv->plaisun;
53  if (epv->plaishade < 0.0)
54  {
55  bgc_printf(BV_ERROR, "FATAL ERROR: Negative plaishade\n");
56  bgc_printf(BV_ERROR, "LAI of shaded canopy = %lf\n",epv->plaishade);
57  ok=0;
58  }
59 
60  /* calculate the projected specific leaf area for sunlit and
61  shaded canopy fractions */
62  epv->sun_proj_sla = (epv->plaisun + (epv->plaishade/epc->sla_ratio)) /
63  cs->leafc;
64  epv->shade_proj_sla = epv->sun_proj_sla * epc->sla_ratio;
65  }
66  else if (cs->leafc == 0.0)
67  {
68  epv->all_lai = 0.0;
69  epv->proj_lai = 0.0;
70  epv->plaisun = 0.0;
71  epv->plaishade = 0.0;
72  epv->sun_proj_sla = 0.0;
73  epv->shade_proj_sla = 0.0;
74  }
75  else
76  {
77  bgc_printf(BV_ERROR, "FATAL ERROR: Negative leaf carbon pool\n");
78  bgc_printf(BV_ERROR, "leafc = %.7e\n",cs->leafc);
79  ok=0;
80  }
81 
82  k = epc->ext_coef;
83  proj_lai = epv->proj_lai;
84 
85  /* calculate total shortwave absorbed */
86  k_sw = k;
87  albedo_sw = albedo;
88 
89  /* FIXED 02/05/04 */
90  if( metv->swavgfd < 0.0 )
91  metv->swavgfd = 0.0;
92 
93  sw = metv->swavgfd * (1.0 - albedo_sw);
94  swabs = sw * (1.0 - exp(-k_sw*proj_lai));
95  swtrans = sw - swabs;
96 
97  /* calculate PAR absorbed */
98  k_par = k * 1.0;
99  albedo_par = albedo/3.0;
100 
101  /* FIXED 02/05/04 */
102  if( metv->par < 0.0 )
103  metv->par = 0.0;
104 
105  par = metv->par * (1.0 - albedo_par);
106  parabs = par * (1.0 - exp(-k_par*proj_lai));
107 
108  /* calculate the total shortwave absorbed by the sunlit and
109  shaded canopy fractions */
110  swabs_plaisun = k_sw * sw * epv->plaisun;
111  swabs_plaishade = swabs - swabs_plaisun;
112 
113  /* FIXED 02/05/04 */
114  if( swabs_plaishade < 0.0 )
115  {
116  swabs_plaisun = swabs;
117  swabs_plaishade = 0;
118  }
119 
120  if (swabs_plaishade < 0.0) /* AAN: Is this block even necesary? */
121  {
122  bgc_printf(BV_ERROR, "FATAL ERROR: negative swabs_plaishade (%lf)\n",swabs_plaishade);
123  ok=0;
124  }
125 
126  /* convert this to the shortwave absorbed per unit LAI in the sunlit and
127  shaded canopy fractions */
128  if (proj_lai > 0.0)
129  {
130  swabs_per_plaisun = swabs_plaisun/epv->plaisun;
131  swabs_per_plaishade = swabs_plaishade/epv->plaishade;
132  }
133  else
134  {
135  swabs_per_plaisun = swabs_per_plaishade = 0.0;
136  }
137 
138  /* calculate the total PAR absorbed by the sunlit and
139  shaded canopy fractions */
140  parabs_plaisun = k_par * par * epv->plaisun;
141  parabs_plaishade = parabs - parabs_plaisun;
142 
143  /* FIXED 02/05/04 */
144  if (parabs_plaishade < 0.0)
145  {
146  parabs_plaisun = parabs;
147  parabs_plaishade = 0.0;
148  }
149 
150  if (parabs_plaishade < 0.0) /* AAN: again, not necesary. */
151  {
152  bgc_printf(BV_ERROR, "FATAL ERROR: negative parabs_plaishade (%lf)\n",parabs_plaishade);
153  ok=0;
154  }
155 
156  /* convert this to the PAR absorbed per unit LAI in the sunlit and
157  shaded canopy fractions */
158  if (proj_lai > 0.0)
159  {
160  parabs_per_plaisun = parabs_plaisun/epv->plaisun;
161  parabs_per_plaishade = parabs_plaishade/epv->plaishade;
162  }
163  else
164  {
165  parabs_per_plaisun = parabs_per_plaishade = 0.0;
166  }
167 
168  /* assign structure values */
169  metv->swabs = swabs;
170  metv->swtrans = swtrans;
171  metv->swabs_per_plaisun = swabs_per_plaisun;
172  metv->swabs_per_plaishade = swabs_per_plaishade;
173  /* calculate PPFD: assumes an average energy for PAR photon (EPAR, umol/J)
174  unit conversion: W/m2 --> umol/m2/s. */
175  metv->ppfd_per_plaisun = parabs_per_plaisun * EPAR;
176  metv->ppfd_per_plaishade = parabs_per_plaishade * EPAR;
177  metv->parabs = parabs;
178 
179  return (!ok);

14 int daily_water_state_update(wflux_struct* wf, wstate_struct* ws)
15 {
16  /* daily update of the water state variables */
17 
18  int ok=1;
19 
20  /* precipitation fluxes */
21  ws->canopyw += wf->prcp_to_canopyw;
22  ws->prcp_src += wf->prcp_to_canopyw;
23  ws->soilw += wf->prcp_to_soilw;
24  ws->prcp_src += wf->prcp_to_soilw;
25  ws->snoww += wf->prcp_to_snoww;
26  ws->prcp_src += wf->prcp_to_snoww;
27 
28  /* snowmelt fluxes */
29  ws->soilw += wf->snoww_to_soilw;
30  ws->snoww -= wf->snoww_to_soilw;
31  ws->snowsubl_snk += wf->snoww_subl;
32  ws->snoww -= wf->snoww_subl;
33 
34  /* bare soil evaporation */
35  ws->soilevap_snk += wf->soilw_evap;
36  ws->soilw -= wf->soilw_evap;
37 
38  /* canopy intercepted water fluxes */
39  ws->canopyevap_snk += wf->canopyw_evap;
40  ws->canopyw -= wf->canopyw_evap;
41  ws->soilw += wf->canopyw_to_soilw;
42  ws->canopyw -= wf->canopyw_to_soilw;
43 
44  /* transpiration */
45  ws->trans_snk += wf->soilw_trans;
46  ws->soilw -= wf->soilw_trans;
47 
48  /* outflow */
49  ws->outflow_snk += wf->soilw_outflow;
50  ws->soilw -= wf->soilw_outflow;
51 
52  /* the following special case prevents evaporation under very
53  dry conditions from causing a negative soilwater content */
54  if (ws->soilw < 0.0) /* negative soilwater */
55  {
56  /* add back the evaporation and transpiration fluxes, and
57  set these fluxes to 0.0 */
58  ws->soilw += wf->soilw_evap;
59  ws->soilevap_snk -= wf->soilw_evap;
60  ws->soilw += wf->soilw_trans;
61  ws->trans_snk -= wf->soilw_trans;
62  wf->soilw_evap = 0.0;
63  wf->soilw_trans = 0.0;
64 
65  /* test again for negative soilwater...should never be true */
66  if (ws->soilw < 0.0)
67  {
68  bgc_printf(BV_ERROR, "Fatal negative soilwater error\n");
69  bgc_printf(BV_ERROR, "soilwater = %e\n",ws->soilw);
70  bgc_printf(BV_ERROR, "most likely a problem with reading the met data input file...\n");
71  ok=0;
72  }
73  }
74 
75  return (!ok);
76 }
